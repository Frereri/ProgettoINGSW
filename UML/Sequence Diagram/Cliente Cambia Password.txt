
@startuml SD_04_Cliente_Cambia_Password
title SD-04: Cliente Cambia Password

actor Cliente as C
participant "ClienteDashboard\n(Frontend)" as FE
participant "ChangePasswordForm\n(Frontend)" as CPF
participant "UtenteController\n/api/utente" as UC
participant "AuthService" as AS
participant "AWS Cognito" as COG

C -> FE : Accede alla dashboard
FE --> C : Mostra dashboard
activate FE

C -> CPF : Clicca "Cambia Password"
activate CPF
CPF -> CPF : Mostra form\n(vecchia password,\nnuova password)

C -> CPF : Compila e conferma
CPF -> UC : POST /api/utente/update-password\nHeader: Authorization: Bearer <accessToken>\nBody: {vecchiaPassword, nuovaPassword}
activate UC

UC -> UC : Estrae accessToken\ndall'header (substring(7))

UC -> AS : updatePassword(\n  accessToken,\n  vecchiaPassword,\n  nuovaPassword)
activate AS

AS -> COG : ChangePasswordRequest(\n  previousPassword,\n  proposedPassword,\n  accessToken)
activate COG

alt Credenziali valide e nuova password conforme
    COG --> AS : 200 OK
    deactivate COG
    AS --> UC : (nessuna eccezione)
    deactivate AS
    UC --> CPF : 200 OK\n"Password aggiornata con successo!"
else Errore (password errata / non conforme)
    COG --> AS : NotAuthorizedException\no InvalidPasswordException
    activate COG
    deactivate COG
    AS --> UC : Exception
    deactivate AS
    UC --> CPF : 400 Bad Request\n"Errore: <messaggio>"
end

deactivate UC
CPF --> C : Mostra messaggio\ndi successo o errore
deactivate CPF
deactivate FE

@enduml


@startuml SD_03_Cliente_Effettua_Offerta
title SD-03: Cliente Effettua Offerta

actor Cliente as C
participant "ClienteDashboard\n(Frontend)" as FE
participant "DettagliImmobile\n(Frontend)" as DI
participant "ClienteController\n/api/cliente" as CC
participant "ClienteService" as CS
participant "ClienteRepo" as CR
participant "ImmobileRepo" as IR
participant "OffertaRepo" as OR
participant "StoricoRepo" as SR
participant "IOffertaMapper" as MAP
database "Database" as DB

C -> FE : Accede alla dashboard
FE --> C : Lista immobili disponibili
activate FE

C -> DI : Seleziona un immobile
activate DI
DI -> DI : Mostra dettagli\ne bottone "Fai un'offerta"

C -> DI : Clicca "Fai un'offerta"\ne inserisce prezzo
DI -> CC : POST /api/cliente/offerte/crea\n{OffertaDTO: idImmobile, prezzoOfferto}\n[JWT Bearer]
activate CC

CC -> CC : Estrae idCliente dal JWT\n(jwt.getSubject())
CC -> CS : inviaOfferta(dto, idCliente)
activate CS

CS -> CR : clienteRepo.findByIdUtente(idCliente)
activate CR
CR -> DB : SELECT Cliente
DB --> CR : Cliente
CR --> CS : Cliente
deactivate CR

CS -> IR : immobileRepo.findById(dto.idImmobile)
activate IR
IR -> DB : SELECT Immobile
DB --> IR : Immobile
IR --> CS : Immobile
deactivate IR

CS -> MAP : dtoToOfferta(dto)
MAP --> CS : Offerta

CS -> CS : setta cliente, immobile\nprezzoOriginale = immobile.prezzo\nstato = "IN_ATTESA"

CS -> OR : offertaRepo.save(offerta)
activate OR
OR -> DB : INSERT Offerta
DB --> OR : Offerta salvata
OR --> CS : Offerta
deactivate OR

CS -> SR : storicoMapper.createLog(\n  offerta, idCliente,\n  "Offerta iniziale creata dal cliente")
CS -> SR : storicoRepo.save(log)
activate SR
SR -> DB : INSERT StoricoOfferta
DB --> SR : Log salvato
SR --> CS : ok
deactivate SR

CS --> CC : OffertaDTO
deactivate CS

CC --> DI : 200 OK {OffertaDTO}
deactivate CC

DI --> C : Mostra conferma\n"Offerta inviata con successo"
deactivate DI
deactivate FE

@enduml
